Supported Attributes
MdlLayout
  • fixedDrawer: fixed drawer. (default: false)
  • drawerTitle: title for the drawer.
  • drawerMenu: MenuItem array for the drawer menu.
MdlHeader
  • title: title for the header.
  • menu: MenuItem array for the header menu.
  • transparent: transparent header. (default: false)
  • noHeader: no header. (default: false)
  • waterfall: waterfall header. (default: false)
MdlDrawer
  • title: title for the drawer.
  • menu: MenuItem array for the drawer menu.
MenuItem Interface

  interface MenuItem {{'{'}}
    title: string; // Title that will appear in the menu for a menu item.
    link: string; // Link for the menu item. Don't need to provide if using routerLink
    routerLink: Array<any>; // routerLink Array for angular 2 router.
  {{'}'}}
